     2  library ieee;
     3  use ieee.std_logic_1164.all;
     4  use ieee.numeric_std.all;
     5  use work.constantspackage.all;
     6  use work.vpfRecords.all;
     7  use work.portspackage.all;
     8  entity gridLockFifo is
     9  generic (
     10      FIFO_DEPTH       : integer := 24;
     11      FIFO_DATA_WIDTH  : integer := 24;
     12      FIFO_ADDR_WIDTH  : integer := 14);
     13  port (
     14      clk              : in  std_logic;
     15      clrStatus        : in  std_logic;
     16      --READ PORT
     17      rdEn             : in  std_logic;
     18      rdAddress        : in  std_logic_vector (FIFO_ADDR_WIDTH-1 downto 0);
     19      dataO            : out std_logic_vector (FIFO_DATA_WIDTH-1 downto 0);
     20      --WRITE PORT
     21      wrEn             : in  std_logic;
     22      wrAddress        : in  std_logic_vector (FIFO_ADDR_WIDTH-1 downto 0);
     23      dataIn           : in  std_logic_vector (FIFO_DATA_WIDTH-1 downto 0);
     24      --STATUS
     25      wrDone           : out std_logic;
     26      rdDone           : out std_logic;
     27      emptyO           : out std_logic;
     28      fullO            : out std_logic);
     29  end entity;
     30  architecture rtl of gridLockFifo is
     31      type RAM is array (integer range <>)of std_logic_vector (FIFO_DATA_WIDTH-1 downto 0);
     32      signal Mem : RAM (0 to FIFO_DEPTH-1);
     33      signal eqlLocations  : std_logic;
     34      signal wrDoneStatus  : std_logic;
     35      signal rdDoneStatus  : std_logic;
     36      signal preStatus     : std_logic := lo;
     37      signal preFull       : std_logic;
     38      signal preEmpty      : std_logic;
     39      signal emptyHi       : std_logic;
     40      signal fullHi        : std_logic;
     41      signal fifoWrAddrs   : integer := 0;
     42      signal fifoRdAddrs   : integer := 0;
     43  begin
     44  
     45  fullO        <= fullHi;
     46  emptyO       <= emptyHi;
     47  
     48  fifoWrAddrs  <= to_integer(unsigned(wrAddress));
     49  fifoRdAddrs  <= to_integer(unsigned(rdAddress));
     50  
     51  eqlLocations <= hi when (wrAddress = rdAddress) else lo;
     52  wrDone       <= hi when (fifoWrAddrs    = FIFO_DEPTH - 1) else lo;
     53  rdDone       <= hi when (fifoRdAddrs    = FIFO_DEPTH - 1) else lo;
     54  preEmpty     <= not preStatus and eqlLocations;
     55  preFull      <= preStatus and eqlLocations;
     56  
     57  readPort: process (clk) begin
     58      if (rising_edge(clk)) then
     59          if (rdEn = hi and emptyHi = lo) then
     60              dataO <= Mem(fifoRdAddrs);
     61          end if;
     62      end if;
     63  end process readPort;
     64  
     65  writePort: process (clk) begin
     66      if (rising_edge(clk)) then
     67          if (wrEn = hi and fullHi = lo) then
     68              Mem(fifoWrAddrs) <= dataIn;
     69          end if;
     70      end if;
     71  end process writePort;
     72  
     73  rdPointerStatus: process (wrAddress, rdAddress)
     74      variable rdConditionBit0 : std_logic;
     75      variable rdConditionBit1 : std_logic;
     76  begin
     77      rdConditionBit0 := wrAddress(FIFO_ADDR_WIDTH-2) xor  rdAddress(FIFO_ADDR_WIDTH-1);
     78      rdConditionBit1 := wrAddress(FIFO_ADDR_WIDTH-1) xnor rdAddress(FIFO_ADDR_WIDTH-2);
     79      rdDoneStatus    <= rdConditionBit0 and rdConditionBit1;
     80  end process rdPointerStatus;
     81  
     82  wrPointerStatus: process (wrAddress, rdAddress)
     83      variable wrConditionBit0 : std_logic;
     84      variable wrConditionBit1 : std_logic;
     85  begin
     86      wrConditionBit0 := wrAddress(FIFO_ADDR_WIDTH-2) xnor rdAddress(FIFO_ADDR_WIDTH-1);
     87      wrConditionBit1 := wrAddress(FIFO_ADDR_WIDTH-1) xor  rdAddress(FIFO_ADDR_WIDTH-2);
     88      wrDoneStatus    <= wrConditionBit0 and wrConditionBit1;
     89  end process wrPointerStatus;
     90  
     91  fifoPreStatus: process (wrDoneStatus, rdDoneStatus, clrStatus) begin
     92      if (rdDoneStatus = hi or clrStatus = hi) then
     93          preStatus <= lo;  -- GoingEmpty
     94      elsif (wrDoneStatus = hi) then
     95          preStatus <= hi;  -- GoingFull
     96      end if;
     97  end process fifoPreStatus;
     98  
     99  fifoFullStatus: process (clk, preFull) begin
     100      if (preFull = hi) then
     101          fullHi <= hi;
     102      elsif (rising_edge(clk)) then
     103          fullHi <= lo;
     104      end if;
     105  end process fifoFullStatus;
     106  
     107  fifoEmptyStatus: process (clk, preEmpty) begin
     108      if (preEmpty = hi) then
     109          emptyHi <= hi;
     110      elsif (rising_edge(clk)) then
     111          emptyHi <= lo;
     112      end if;
     113  end process fifoEmptyStatus;
     114  
     115  end architecture;
